7 class RuleKitJavaException(Exception):
8 """Wrapper method for handling Java exceptions produced by RuleKit.
9 Useful for debugging internal Java code error. It allows to access java
10 stack traces easily.
11  
12 Example:
13 >>> from rulekit import RuleKit
14 >>> from rulekit.classification import RuleClassifier
15 >>>
16 >>> clf = RuleClassifier()
17 >>> try:
18 >>> clf.fit(X, y)
19 >>> except JException as e: # catch java exceptions
20 >>> e = RuleKitJavaException(e) # wrap them for easier handling
21 >>> e.print_java_stack_trace() # access java stack trace
22 """
